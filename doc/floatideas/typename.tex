\section{Proposal}
Quick glance:
\begin{codeblock}
// declare some materials
struct Color {
  void apply() {}
};

struct Texture {
  void apply() {}
};

struct Glass {
  void apply() {}
};

// define a type alias
typename Material {
  void apply();
};

int main() {
  {
    // use Material as a pointer
    Color color;
    Material* material = color;
    material->apply();
  }
  {
    // use Material as a reference
    Texture texture;
    Material& material = texture;
    material.apply();
  }
  {
    // host a Material in unique ptr
    std::unique_ptr<Material> material{new Glass()};
    material->apply();
  }
}
\end{codeblock}
